import sys


def parse_fasta(file):
    sequence_name = ""
    sequence = ""
    sequences = {}
    for line in file:
        if line.startswith(">"):
            if sequence_name:
                sequences[sequence_name] = sequence
            sequence_name = line.strip()[1:]
            sequence = ""
        else:
            sequence += line.strip()
    sequences[sequence_name] = sequence
    return sequences


def main():
    file = sys.argv[1]
    with open(file) as f:
        lines = f.read().splitlines()
    sequences = parse_fasta(lines)

    for s in sequences:
        for t in sequences:
            if s == t:
                continue
            if sequences[s][-3:] == sequences[t][:3]:
                print(s, t)


if __name__ == "__main__":
    main()
